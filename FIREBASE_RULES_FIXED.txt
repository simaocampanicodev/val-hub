rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Acesso à dashboard completa (alterar roles, etc.)
    function hasDashboardAccess() {
      return request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && (
          get(/databases/$(database)/documents/users/$(request.auth.uid)).data.username == 'txger.'
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['owner', 'mod', 'dev']
        );
    }

    // Acesso aos tickets (inclui helpers – só leitura de tickets)
    function canReadTickets() {
      return request.auth != null
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && (
          get(/databases/$(database)/documents/users/$(request.auth.uid)).data.username == 'txger.'
          || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['owner', 'mod', 'dev', 'helper']
        );
    }

    match /users/{userId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow delete: if false;
      allow update: if request.auth != null && (
        // ✅ NOVO: Permite atualizar campos de pontos quando é o próprio utilizador
        (request.auth.uid == userId && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'points', 'xp', 'level', 'wins', 'losses', 'winstreak', 
          'lastPointsChange', 'lastSeenAt', 'reputation', 'avatarUrl', 'bannerUrl',
          'riotId', 'riotTag', 'active_quests', 'lastDailyQuestGeneration', 'lastMonthlyQuestGeneration'
        ]))
        // Adicional: Permite atualizar friends/friend_requests
        || request.resource.data.diff(resource.data).affectedKeys().hasOnly(['friend_requests', 'friends'])
        // Admin pode alterar role e verified
        || (
          request.resource.data.diff(resource.data).affectedKeys().hasOnly(['role', 'verified'])
          && hasDashboardAccess()
        )
      );
    }

    match /queue_entries/{entryId} {
      allow read: if true;
      allow create: if request.auth != null && request.auth.uid == entryId;
      allow update: if request.auth != null && request.auth.uid == entryId;
      allow delete: if request.auth != null;
    }

    match /active_matches/{matchId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update: if request.auth != null;
      allow delete: if request.auth != null;
    }

    match /matches/{matchId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if false;
    }

    match /tickets/{ticketId} {
      allow read: if canReadTickets();
      allow create: if request.auth != null;
      allow update, delete: if false;
    }
  }
}
